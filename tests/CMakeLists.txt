# Basic tests
set(PYBIAS_LIBRARY_DIR ${CMAKE_BINARY_DIR}/src)

file(GLOB BASIC_TEST_NAMES RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} basic_*)
list(APPEND TEST_NAMES ${BASIC_TEST_NAMES})

file(GLOB BIAS_TEST_NAMES RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} bias_*)
list(APPEND TEST_NAMES ${BIAS_TEST_NAMES})

foreach(TEST_NAME ${TEST_NAMES})
  add_test(NAME ${TEST_NAME} COMMAND test.sh ${PYBIAS_LIBRARY_DIR}
           WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/${TEST_NAME})
endforeach(TEST_NAME)
